# 9. BLUETOOTH ПЕЧАТ (DTM-5811)

## 9.1 Setup

```javascript
async function pairPrinter() {
    const device = await navigator.bluetooth.requestDevice({
        filters: [{ name: 'DTM-5811' }],
        optionalServices: ['000018f0-0000-1000-8000-00805f9b34fb']
    });
    const server = await device.gatt.connect();
    const service = await server.getPrimaryService('000018f0-0000-1000-8000-00805f9b34fb');
    const characteristic = await service.getCharacteristic('00002af1-0000-1000-8000-00805f9b34fb');
    localStorage.setItem('printer_device_id', device.id);
    return characteristic;
}
```

## 9.2 TSPL команди

```javascript
function generateTSPL(product, store) {
    return `
SIZE 50 mm, 30 mm
GAP 2 mm, 0 mm
DIRECTION 1
REFERENCE 0,0
CLS
TEXT 10,10,"3",0,1,1,"${store.name}"
TEXT 10,40,"2",0,1,1,"${product.name}"
BARCODE 10,70,"128",70,1,0,2,2,"${product.barcode}"
TEXT 10,155,"3",0,1,1,"${product.price_eur} EUR"
TEXT 10,185,"2",0,1,1,"${product.price_bgn} лв"
PRINT 1
`;
}

async function printLabel(product) {
    const characteristic = await getCharacteristic();
    const tspl = generateTSPL(product, currentStore);
    const encoder = new TextEncoder();
    await characteristic.writeValue(encoder.encode(tspl));
}
```
